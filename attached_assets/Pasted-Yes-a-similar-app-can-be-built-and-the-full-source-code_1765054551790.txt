import React, { useState } from 'react';

const OVERPASS_URL = 'https://overpass-api.de/api/interpreter';

function haversineDistanceKm(lat1, lon1, lat2, lon2) {
  const toRad = (d) => (d * Math.PI) / 180;
  const R = 6371;
  const dLat = toRad(lat2 - lat1);
  const dLon = toRad(lon2 - lon1);
  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) *
      Math.cos(toRad(lat2)) *
      Math.sin(dLon / 2) *
      Math.sin(dLon / 2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c;
}

export default function App() {
  const [lat, setLat] = useState('');
  const [lon, setLon] = useState('');
  const [radiusKm, setRadiusKm] = useState(20);
  const [brandFilter, setBrandFilter] = useState('all');
  const [loading, setLoading] = useState(false);
  const [stations, setStations] = useState([]);
  const [error, setError] = useState('');

  const handleUseMyLocation = () => {
    if (!navigator.geolocation) {
      setError('Geolocation is not supported by this browser.');
      return;
    }
    setError('');
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        setLat(pos.coords.latitude.toFixed(6));
        setLon(pos.coords.longitude.toFixed(6));
      },
      (err) => {
        setError('Unable to get location: ' + err.message);
      },
      { enableHighAccuracy: true }
    );
  };

  const buildOverpassQuery = (latValue, lonValue, radiusMeters) => {
    return `
      [out:json];
      (
        node
          ["amenity"="fuel"]
          (around:${radiusMeters},${latValue},${lonValue});
      );
      out center;
    `;
  };

  const fetchStations = async () => {
    setError('');
    setStations([]);
    if (!lat || !lon) {
      setError('Please provide a location or use the location button.');
      return;
    }
    setLoading(true);
    try {
      const radiusMeters = radiusKm * 1000;
      const query = buildOverpassQuery(lat, lon, radiusMeters);
      const res = await fetch(OVERPASS_URL, {
        method: 'POST',
        body: query,
        headers: {
          'Content-Type': 'text/plain;charset=UTF-8'
        }
      });
      if (!res.ok) {
        throw new Error('Overpass API error: ' + res.status);
      }
      const data = await res.json();
      const elements = data.elements || [];
      const parsed = elements
        .filter((el) => el.type === 'node')
        .map((el) => {
          const name = el.tags?.name || 'Unnamed fuel station';
          const brand = el.tags?.brand || 'Unknown brand';
          const distanceKm = haversineDistanceKm(
            parseFloat(lat),
            parseFloat(lon),
            el.lat,
            el.lon
          );
          return {
            id: el.id,
            name,
            brand,
            lat: el.lat,
            lon: el.lon,
            distanceKm
          };
        })
        .sort((a, b) => a.distanceKm - b.distanceKm);

      const filtered =
        brandFilter === 'all'
          ? parsed
          : parsed.filter(
              (s) =>
                s.brand.toLowerCase() === brandFilter.toLowerCase()
            );

      setStations(filtered);
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  };

  const directionsUrl = (station) =>
    `https://www.openstreetmap.org/directions?engine=fossgis_osrm_car&route=${lat},${lon};${station.lat},${station.lon}`;

  return (
    <div className="app-root">
      <header className="header">
        <h1>Fuel Finder</h1>
        <p>‚õΩ Find fuel stations near you</p>
        <nav className="nav-links">
          <a href="#about">About</a>
          <a href="#terms">Terms</a>
          <a href="#privacy">Privacy</a>
        </nav>
      </header>

      <main className="main">
        <section className="card">
          <h2>Need to refuel?</h2>
          <p>Let us find fuel stations within 20 km of you.</p>

          <div className="form-group">
            <label>Location</label>
            <div className="location-row">
              <div className="location-inputs">
                <input
                  type="number"
                  placeholder="Latitude"
                  value={lat}
                  onChange={(e) => setLat(e.target.value)}
                />
                <input
                  type="number"
                  placeholder="Longitude"
                  value={lon}
                  onChange={(e) => setLon(e.target.value)}
                />
              </div>
              <button
                type="button"
                className="btn-secondary"
                onClick={handleUseMyLocation}
              >
                Use My Location üìç
              </button>
            </div>
            <small>
              For now, latitude/longitude are used instead of ZIP code.
            </small>
          </div>

          <div className="form-row">
            <div className="form-group">
              <label>Radius</label>
              <select
                value={radiusKm}
                onChange={(e) =>
                  setRadiusKm(Number(e.target.value))
                }
              >
                <option value={5}>5 km</option>
                <option value={10}>10 km</option>
                <option value={20}>20 km</option>
                <option value={50}>50 km</option>
              </select>
            </div>

            <div className="form-group">
              <label>Brand</label>
              <select
                value={brandFilter}
                onChange={(e) => setBrandFilter(e.target.value)}
              >
                <option value="all">All Brands</option>
                <option value="Shell">Shell</option>
                <option value="BP">BP</option>
                <option value="Total">Total</option>
              </select>
            </div>
          </div>

          <button
            type="button"
            className="btn-primary"
            onClick={fetchStations}
            disabled={loading}
          >
            {loading ? 'Searching‚Ä¶' : 'Find Stations'}
          </button>

          {error && <p className="error">{error}</p>}
        </section>

        <section className="results">
          <h2>Results</h2>
          {stations.length === 0 && !loading && (
            <p>No stations found yet. Run a search.</p>
          )}
          <ul className="station-list">
            {stations.map((s) => (
              <li key={s.id} className="station-item">
                <div>
                  <div className="station-name">{s.name}</div>
                  <div className="station-meta">
                    <span>{s.brand}</span> ¬∑{' '}
                    <span>
                      {s.distanceKm.toFixed(1)} km away
                    </span>
                  </div>
                </div>
                <a
                  className="btn-link"
                  href={directionsUrl(s)}
                  target="_blank"
                  rel="noreferrer"
                >
                  üß≠ Directions
                </a>
              </li>
            ))}
          </ul>
        </section>
      </main>

      <footer className="footer">
        <span id="about">About</span> ‚Ä¢
        <span id="terms">Terms</span> ‚Ä¢
        <span id="privacy">Privacy</span>
      </footer>
    </div>
  );
}

import React from 'react';
import { MapContainer, TileLayer, Marker, Popup, Circle } from 'react-leaflet';
import L from 'leaflet';

// Simple default icon fix for modern bundlers
const DefaultIcon = L.icon({
  iconUrl:
    'https://unpkg.com/leaflet@1.9.4/dist/images/marker-icon.png',
  shadowUrl:
    'https://unpkg.com/leaflet@1.9.4/dist/images/marker-shadow.png',
  iconSize: [25, 41],
  iconAnchor: [12, 41]
});
L.Marker.prototype.options.icon = DefaultIcon;

export default function MapView({ center, radiusKm, stations }) {
  if (!center) return null;

  const [lat, lon] = center;

  return (
    <div className="map-wrapper">
      <MapContainer
        center={[lat, lon]}
        zoom={12}
        scrollWheelZoom={true}
        style={{ height: '400px', width: '100%' }}
      >
        <TileLayer
          attribution='&copy; OpenStreetMap contributors'
          url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
        />
        <Circle
          center={[lat, lon]}
          radius={radiusKm * 1000}
          pathOptions={{ color: '#22c55e', fillOpacity: 0.1 }}
        />
        <Marker position={[lat, lon]}>
          <Popup>Your location</Popup>
        </Marker>

        {stations.map((s) => (
          <Marker key={s.id} position={[s.lat, s.lon]}>
            <Popup>
              <strong>{s.name}</strong>
              <br />
              {s.brand}
              <br />
              {s.distanceKm.toFixed(1)} km away
            </Popup>
          </Marker>
        ))}
      </MapContainer>
    </div>
  );
}
